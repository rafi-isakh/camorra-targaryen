  $(document).ready(function() {
    $("#jumlahpendanaan").autoNumeric('init',{aSep:'.',aDec:',', vMin: '0', vMax: '999999999'});
    $("#total_lend").autoNumeric('init',{aSep:'.',aDec:',', vMin: '0', vMax: '999999999'});
    $("#total_lend").autoNumeric('set', 0);
    document.getElementById("jumlahpendanaan").focus();
    document.getElementById("total_lend").focus();
    document.getElementById("jumlahpendanaan").blur();
    document.getElementById("total_lend").blur();
    $('#jumlahpendanaan').keyup(function () {
      updateInput();
    });
  });
  function updateInput(){
      var fund = document.getElementsByName("fund")[0].value;
      var values = fund.replace(/\./g, "");
          fund = parseFloat(values.substring(3));
      var choice = document.getElementsByName("choice")[0].value;
      var grade = document.getElementsByName("grade")[0].value;
      var grade_percent = 1
      switch(grade) {
      case "A":
          if (choice == "true"){
            grade_percent = 19/100;
          }else{
            grade_percent = 10.8/100;
          }
          break;
      case "B":
          if (choice == "true"){
            grade_percent = 24/100;
          }else{
            grade_percent = 13.56/100;
          }
          break;
      case "C":
          if (choice == "true"){
            grade_percent = 31/100;
          }else{
            grade_percent = 16.8/100;
          }
          break;
      default:
      }
      total = (fund * (grade_percent/12) * 12 ) + fund;
      if (isNaN(total)) total = 0;
        $("#total_lend").autoNumeric('set', total);
    }

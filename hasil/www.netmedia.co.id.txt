  	$(document).ready(function() {
		$( "#searchbox" ).keypress(function(e) {
			if(e.keyCode==13 || e.which==13)
			{
				var keyword	= $("#searchbox").val();
				search(keyword);
			}
		});
	});

	function search(keyword)
	{
			
			$('#ll').show();
		$.ajax({
			type    : "POST",
			url     : "http://www.netmedia.co.id/homepage/get_search",
			data    : {keyword: keyword},
			success : function(result) 
			{
						
			$('#ll').hide();
				//$('#skey').html("Search result for "+keyword);
				//$('#found').html(" Video Found");
				$('.searchbox_result').html("");
				$('.searchbox_result').append(result);

				var nice = $("html").niceScroll();  // The document page (body)
				$("#boxscroll").niceScroll({cursorborder:"",cursorcolor:"#2b323a",boxzoom:true,railpadding:{top:0,right:10,left:0,bottom:0}});

				$('.detail-pop').magnificPopup({
					type: 'ajax',
					alignTop: true,
					overflowY: 'scroll', // as we know that popup content is tall we set scroll overflow by default to avoid jump
				}); 
			}
		});
	}

  	$(document).ready(function() {
		$( "#searchbox" ).keypress(function(e) {
			if(e.keyCode==13 || e.which==13)
			{
				var keyword	= $("#searchbox").val();
				search(keyword);
			}
		});
	});

	function search(keyword)
	{
			
			$('#ll').show();
		$.ajax({
			type    : "POST",
			url     : "http://www.netmedia.co.id/homepage/get_search",
			data    : {keyword: keyword},
			success : function(result) 
			{
						
			$('#ll').hide();
				//$('#skey').html("Search result for "+keyword);
				//$('#found').html(" Video Found");
				$('.searchbox_result').html("");
				$('.searchbox_result').append(result);

				var nice = $("html").niceScroll();  // The document page (body)
				$("#boxscroll").niceScroll({cursorborder:"",cursorcolor:"#2b323a",boxzoom:true,railpadding:{top:0,right:10,left:0,bottom:0}});

				$('.detail-pop').magnificPopup({
					type: 'ajax',
					alignTop: true,
					overflowY: 'scroll', // as we know that popup content is tall we set scroll overflow by default to avoid jump
				}); 
			}
		});
	}
